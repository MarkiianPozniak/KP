import 'package:flutter/material.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        backgroundColor: Colors.white,
        body: Center(
          child: ExampleStack(),
        ),
      ),
    );
  }
}

class ExampleStack extends StatelessWidget {
  final double a = 24; // жовта рамка
  final double b = 34; // червона рамка
  final double c = 64; // синя область

  @override
  Widget build(BuildContext context) {
    return Container(
      width: 350,
      height: 200,
      color: Colors.grey[200],
      child: Stack(
        children: [
          // ЖОВТИЙ
          Positioned(
            left: 0,
            top: 0,
            child: Container(
              width: 350,
              height: 200,
              color: Colors.yellow,
            ),
          ),

          // ЧЕРВОНИЙ
          Positioned(
            left: a,
            top: a,
            child: Container(
              width: 350 - a,
              height: 200 - a,
              color: Colors.red,
            ),
          ),

          // СИНІЙ
          Positioned(
            left: a + b,
            top: a + b,
            child: Container(
              width: 350 - (a + b),
              height: 200 - (a + b),
              color: Colors.blue,
            ),
          ),

          // Текст "Hello"
          Positioned(
            left: 175,
            top: a + 10,
            child: RichText(
              text: TextSpan(
                text: 'Hello',
                style: TextStyle(
                  fontSize: 18,
                  color: Colors.white,
                  fontWeight: FontWeight.bold,
                ),
              ),
            ),
          ),

          // Текст "Flutter"
          Positioned(
            left: a + b + 235,
            bottom: 1,
            child: RichText(
              text: TextSpan(
                text: 'Flutter',
                style: TextStyle(
                  fontSize: 18,
                  color: Colors.white,
                  fontWeight: FontWeight.bold,
                ),
              ),
            ),
          ),
        ],
      ),
    );
  }
}
